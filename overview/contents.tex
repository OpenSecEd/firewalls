    \begin{example}[Firewall]
      \begin{itemize}

      \end{itemize}

  }

\begin{frame}

      \begin{itemize}

        \item Internal and going outside.
      \end{itemize}
    \end{definition}
  }
  \only<2>{
    \begin{example}[Network-based intrusion detection]
      \begin{itemize}
        \item Connection from laptop to server.
        \item Immediately followed by connection to outside.
      \end{itemize}
    \end{example}
  }
\end{frame}
